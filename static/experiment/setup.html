<script>
  var experiment = new abcd.Experiment('{{experiment.id}}');

  experiment.participate(['control', 'blue button', 'red button'], function (error, result) {
    if (error) throw error;

    if (result.variant === 'control') {
      // Do nothing
    } else if (result.variant === 'blue button') {
      $('button').css({
        background: 'blue'
      });
    } else if (result.variant === 'red button') {
      $('button').css({
        background: 'red'
      });
    }

    $('button').on('click', function () {
      experiment.convert();
    });
  });
</script>
